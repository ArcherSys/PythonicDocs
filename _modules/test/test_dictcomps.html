        with self.assertRaisesRegex(SyntaxError, "can't assign"):
            compile("{x: y for y, x in ((1, 2), (3, 4))} = 5", "<test>",
                    "exec")

        with self.assertRaisesRegex(SyntaxError, "can't assign"):
