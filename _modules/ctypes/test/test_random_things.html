[docs]    def capture_stderr(self, func, *args, **kw):
        # helper - call function 'func', and return the captured stderr
        import io
        old_stderr = sys.stderr
        logger = sys.stderr = io.StringIO()
        try:
            func(*args, **kw)
        finally:
            sys.stderr = old_stderr
        return logger.getvalue()
